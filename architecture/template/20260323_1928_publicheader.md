<script setup>
import { ref, computed, onMounted, onBeforeUnmount } from 'vue'
import { router, usePage, Link } from '@inertiajs/vue3'
import { useI18n } from 'vue-i18n'

const props = defineProps({
  locale: { type: String, default: null },
  disableLanguageSelector: { type: Boolean, default: false },
  breadcrumbs: { type: Array, default: () => [] },
  electionMode: { type: Boolean, default: false }, // ← NEW: switches font style
})

const { t, locale } = useI18n()
const page = usePage()

const currentLocale = ref(getInitialLocale())
const showMobileMenu = ref(false)
const isLoggingOut = ref(false)

const isLoggedIn = computed(() => !!page.props.user)
const breadcrumbs = computed(() => props.breadcrumbs || page.props.breadcrumbs || [])

function getInitialLocale() {
  const saved = localStorage.getItem('preferred_locale')
  if (saved && ['de', 'en', 'np'].includes(saved)) return saved
  if (props.locale && ['de', 'en', 'np'].includes(props.locale)) return props.locale
  if (locale?.value && ['de', 'en', 'np'].includes(locale.value)) return locale.value
  return 'de'
}

function handleLanguageChange(event) {
  const newLocale = event.target.value
  currentLocale.value = newLocale
  
  // Update i18n
  if (locale) locale.value = newLocale
  
  // Save preference
  localStorage.setItem('preferred_locale', newLocale)
  
  // Set cookie for backend
  const date = new Date()
  date.setFullYear(date.getFullYear() + 1)
  document.cookie = `locale=${newLocale}; expires=${date.toUTCString()}; path=/`
}

function toggleMobileMenu() {
  showMobileMenu.value = !showMobileMenu.value
}

function logout() {
  isLoggingOut.value = true
  router.post(route('logout'), {}, {
    onFinish: () => { isLoggingOut.value = false }
  })
}

// Escape key handler
onMounted(() => {
  document.addEventListener('keydown', (e) => {
    if (e.key === 'Escape' && showMobileMenu.value) showMobileMenu.value = false
  })
})

onBeforeUnmount(() => {
  document.removeEventListener('keydown', () => {})
})
</script>
