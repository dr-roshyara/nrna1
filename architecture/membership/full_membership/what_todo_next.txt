```php
// database/migrations/xxxx_create_membership_payments_table.php
Schema::create('membership_payments', function (Blueprint $table) {
    $table->uuid('id')->primary();
    $table->uuid('member_id');
    $table->uuid('fee_id')->nullable();
    $table->uuid('organisation_id');
    $table->decimal('amount', 10, 2);
    $table->string('currency')->default('EUR');
    $table->string('payment_method'); // bank_transfer, cash, card
    $table->string('transaction_id')->nullable();
    $table->string('status')->default('pending'); // pending, completed, failed
    $table->uuid('recorded_by');
    $table->timestamp('paid_at')->nullable();
    $table->text('notes')->nullable();
    $table->timestamps();
    
    $table->foreign('member_id')->references('id')->on('members');
    $table->foreign('fee_id')->references('id')->on('membership_fees');
    $table->foreign('organisation_id')->references('id')->on('organisations');
    $table->foreign('recorded_by')->references('id')->on('users');
});
```

```php
// app/Models/MembershipPayment.php
class MembershipPayment extends Model
{
    use HasUuids;
    
    protected $fillable = [
        'member_id', 'fee_id', 'organisation_id',
        'amount', 'currency', 'payment_method',
        'transaction_id', 'status', 'recorded_by',
        'paid_at', 'notes',
    ];
    
    protected $casts = [
        'amount' => 'decimal:2',
        'paid_at' => 'datetime',
    ];
    
    public function member(): BelongsTo
    {
        return $this->belongsTo(Member::class);
    }
    
    public function fee(): BelongsTo
    {
        return $this->belongsTo(MembershipFee::class);
    }
    
    public function recordedBy(): BelongsTo
    {
        return $this->belongsTo(User::class, 'recorded_by');
    }
}
```

#### Step 2.2: Create MembershipFinanceController (3 hours)

```php
// app/Http/Controllers/Finance/MembershipFinanceController.php
namespace App\Http\Controllers\Finance;

use App\Http\Controllers\Controller;
use App\Models\Member;
use App\Models\MembershipFee;
use App\Models\MembershipPayment;
use App\Models\Organisation;
use Illuminate\Http\Request;
use Inertia\Inertia;

class MembershipFinanceController extends Controller
{
    public function dashboard(Organisation $organisation)
    {
        $this->authorize('viewFinance', $organisation);
        
        $stats = [
            'total_outstanding' => MembershipFee::where('organisation_id', $organisation->id)
                ->whereIn('status', ['pending', 'overdue'])
                ->sum('amount'),
            'collected_this_month' => MembershipPayment::where('organisation_id', $organisation->id)
                ->whereMonth('paid_at', now()->month)
                ->where('status', 'completed')
                ->sum('amount'),
            'overdue_count' => MembershipFee::where('organisation_id', $organisation->id)
                ->where('status', 'overdue')
                ->count(),
            'pending_payments' => MembershipPayment::where('organisation_id', $organisation->id)
                ->where('status', 'pending')
                ->count(),
        ];
        
        $recentPayments = MembershipPayment::where('organisation_id', $organisation->id)
            ->with(['member.organisationUser.user', 'recordedBy'])
            ->latest()
            ->limit(10)
            ->get();
            
        $overdueMembers = Member::where('organisation_id', $organisation->id)
            ->where('fees_status', 'overdue')
            ->with('organisationUser.user')
            ->limit(10)
            ->get();
        
        return Inertia::render('Finance/Membership/Dashboard', [
            'organisation' => $organisation,
            'stats' => $stats,
            'recentPayments' => $recentPayments,
            'overdueMembers' => $overdueMembers,
        ]);
    }
    
    public function recordPayment(Organisation $organisation, Member $member)
    {
        $this->authorize('manageFinance', $organisation);
        
        $pendingFees = $member->fees()
            ->whereIn('status', ['pending', 'overdue'])
            ->get();
            
        return Inertia::render('Finance/Membership/RecordPayment', [
            'organisation' => $organisation,
            'member' => $member->load('organisationUser.user'),
            'pendingFees' => $pendingFees,
        ]);
    }
    
    public function storePayment(Request $request, Organisation $organisation, Member $member)
    {
        $this->authorize('manageFinance', $organisation);
        
        $validated = $request->validate([
            'fee_id' => 'required|exists:membership_fees,id',
            'amount' => 'required|numeric|min:0.01',
            'payment_method' => 'required|in:bank_transfer,cash,card,other',
            'transaction_id' => 'nullable|string',
            'notes' => 'nullable|string',
        ]);
        
        $payment = MembershipPayment::create([
            'member_id' => $member->id,
            'fee_id' => $validated['fee_id'],
            'organisation_id' => $organisation->id,
            'amount' => $validated['amount'],
            'currency' => 'EUR',
            'payment_method' => $validated['payment_method'],
            'transaction_id' => $validated['transaction_id'],
            'status' => 'completed',
            'recorded_by' => auth()->id(),
            'paid_at' => now(),
            'notes' => $validated['notes'],
        ]);
        
        // Update fee status
        $fee = MembershipFee::find($validated['fee_id']);
        $fee->update(['status' => 'paid', 'paid_at' => now()]);
        
        // Update member fees_status if no pending fees remain
        $hasPendingFees = $member->fees()
            ->whereIn('status', ['pending', 'overdue'])
            ->exists();
            
        if (!$hasPendingFees) {
            $member->update(['fees_status' => 'paid']);
        }
        
        return redirect()
            ->route('finance.membership.dashboard', $organisation->slug)
            ->with('success', 'Payment recorded successfully.');
    }
}
```

### Week 3: Invoice Generation

#### Step 5.1: InvoiceService (3 hours)

```php
// app/Services/InvoiceService.php
namespace App\Services;

use App\Models\MembershipFee;
use App\Models\MembershipPayment;
use Barryvdh\DomPDF\Facade\Pdf;
use Illuminate\Support\Facades\Storage;

class InvoiceService
{
    public function generateForFee(MembershipFee $fee): string
    {
        $member = $fee->member;
        $organisation = $fee->organisation;
        
        $invoiceNumber = sprintf(
            'INV-%s-%04d',
            now()->format('Ymd'),
            $fee->id
        );
        
        $pdf = Pdf::loadView('invoices.membership-fee', [
            'fee' => $fee,
            'member' => $member,
            'organisation' => $organisation,
            'invoiceNumber' => $invoiceNumber,
        ]);
        
        $path = "invoices/{$organisation->id}/{$invoiceNumber}.pdf";
        Storage::put($path, $pdf->output());
        
        $fee->update(['invoice_number' => $invoiceNumber]);
        
        return $path;
    }
    
    public function generateReceipt(MembershipPayment $payment): string
    {
        $receiptNumber = sprintf('RCP-%s-%04d', now()->format('Ymd'), $payment->id);
        
        $pdf = Pdf::loadView('invoices.payment-receipt', [
            'payment' => $payment,
            'receiptNumber' => $receiptNumber,
        ]);
        
        $path = "receipts/{$payment->organisation_id}/{$receiptNumber}.pdf";
        Storage::put($path, $pdf->output());
        
        return $path;
    }
}
```
