# 🏛️ **FINAL ARCHITECTURE: Election Membership System**

This architecture provides a **bulletproof solution** for managing voters across multiple elections while ensuring **data integrity**, **scalability**, and **maintainability**. The design handles 50,000+ voters per election with real-time performance.

## **📊 DATABASE SCHEMA**

### **Core Tables Structure**

```sql
-- =====================================================
-- TABLE: user_organisation_roles (EXISTING)
-- =====================================================
-- This is your source of truth for organisation membership
CREATE TABLE user_organisation_roles (
    user_id UUID NOT NULL,
    organisation_id UUID NOT NULL,
    role VARCHAR(50) NOT NULL DEFAULT 'member',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    
    PRIMARY KEY (user_id, organisation_id),
    FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE,
    FOREIGN KEY (organisation_id) REFERENCES organisations(id) ON DELETE CASCADE,
    
    INDEX idx_organisation_users (organisation_id, user_id)
);

-- =====================================================
-- TABLE: elections (EXISTING - MODIFIED)
-- =====================================================
ALTER TABLE elections ADD UNIQUE KEY unique_org_election (id, organisation_id);

-- Final structure:
CREATE TABLE elections (
    id UUID PRIMARY KEY,
    organisation_id UUID NOT NULL,
    title VARCHAR(255) NOT NULL,
    description TEXT,
    start_date DATE,
    end_date DATE,
    status ENUM('draft', 'active', 'completed', 'archived') DEFAULT 'draft',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    
    FOREIGN KEY (organisation_id) REFERENCES organisations(id) ON DELETE CASCADE,
    UNIQUE KEY unique_org_election (id, organisation_id),  -- For composite FK
    INDEX idx_org_status (organisation_id, status),
    INDEX idx_dates (start_date, end_date)
);

-- =====================================================
-- TABLE: election_memberships (NEW - The Heart of the System)
-- =====================================================
CREATE TABLE election_memberships (
    -- Primary Key
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    
    -- Core References (denormalized for integrity)
    user_id UUID NOT NULL,
    organisation_id UUID NOT NULL,
    election_id UUID NOT NULL,
    
    -- Membership Details
    role ENUM('voter', 'candidate', 'observer', 'admin') NOT NULL DEFAULT 'voter',
    status ENUM('invited', 'active', 'inactive', 'removed') NOT NULL DEFAULT 'active',
    
    -- Metadata
    assigned_by UUID,  -- User who created this membership
    assigned_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    expires_at TIMESTAMP NULL,  -- Optional: voter eligibility expiration
    last_activity_at TIMESTAMP NULL,
    
    -- Flexible Data
    metadata JSON NULL,  -- Store additional attributes
    
    -- Audit Trail
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    
    -- =====================================================
    -- CRITICAL: Composite Foreign Keys
    -- These ensure voter belongs to same organisation as election
    -- =====================================================
    
    -- Ensures user is ACTUALLY a member of this organisation
    CONSTRAINT fk_membership_user_org 
        FOREIGN KEY (user_id, organisation_id) 
        REFERENCES user_organisation_roles(user_id, organisation_id)
        ON DELETE CASCADE,
    
    -- Ensures election belongs to this organisation
    CONSTRAINT fk_membership_election_org 
        FOREIGN KEY (election_id, organisation_id) 
        REFERENCES elections(id, organisation_id)
        ON DELETE CASCADE,
    
    -- Ensures assigned_by is a valid user
    CONSTRAINT fk_membership_assigned_by
        FOREIGN KEY (assigned_by)
        REFERENCES users(id)
        ON DELETE SET NULL,
    
    -- =====================================================
    -- Business Rule Constraints
    -- =====================================================
    
    -- User can have only ONE role per election
    CONSTRAINT unique_user_election UNIQUE (user_id, election_id),
    
    -- Prevent future expiration dates in the past
    CONSTRAINT check_expiration_future 
        CHECK (expires_at IS NULL OR expires_at > created_at),
    
    -- Indexes for Performance
    INDEX idx_election_role (election_id, role, status),
    INDEX idx_user_memberships (user_id, status),
    INDEX idx_org_memberships (organisation_id, role),
    INDEX idx_expiring (expires_at) WHERE expires_at IS NOT NULL,
    INDEX idx_assigned (assigned_by, assigned_at)
    
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;
```

```php
<?php

namespace App\Models;

use Illuminate\Database\Eloquent\Concerns\HasUuids;
use Illuminate\Database\Eloquent\Model;
use Illuminate\Database\Eloquent\Relations\BelongsTo;
use Illuminate\Support\Facades\Cache;
use Illuminate\Support\Facades\DB;

class ElectionMembership extends Model
{
    use HasUuids;

    protected $table = 'election_memberships';

    protected $fillable = [
        'user_id',
        'organisation_id',
        'election_id',
        'role',
        'status',
        'assigned_by',
        'assigned_at',
        'expires_at',
        'last_activity_at',
        'metadata'
    ];

    protected $casts = [
        'assigned_at' => 'datetime',
        'expires_at' => 'datetime',
        'last_activity_at' => 'datetime',
        'metadata' => 'array',
        'created_at' => 'datetime',
        'updated_at' => 'datetime'
    ];

    protected $attributes = [
        'role' => 'voter',
        'status' => 'active',
        'metadata' => '{}'
    ];

    // =====================================================
    // RELATIONSHIPS
    // =====================================================

    public function user(): BelongsTo
    {
        return $this->belongsTo(User::class);
    }

    public function organisation(): BelongsTo
    {
        return $this->belongsTo(Organisation::class);
    }

    public function election(): BelongsTo
    {
        return $this->belongsTo(Election::class);
    }

    public function assigner(): BelongsTo
    {
        return $this->belongsTo(User::class, 'assigned_by');
    }

    // =====================================================
    // SCOPES
    // =====================================================

    public function scopeActive($query)
    {
        return $query->where('status', 'active');
    }

    public function scopeVoters($query)
    {
        return $query->where('role', 'voter');
    }

    public function scopeCandidates($query)
    {
        return $query->where('role', 'candidate');
    }

    public function scopeForElection($query, $electionId)
    {
        return $query->where('election_id', $electionId);
    }

    public function scopeForOrganisation($query, $organisationId)
    {
        return $query->where('organisation_id', $organisationId);
    }

    public function scopeEligible($query)
    {
        return $query->where('status', 'active')
            ->where(function ($q) {
                $q->whereNull('expires_at')
                  ->orWhere('expires_at', '>', now());
            });
    }

    // =====================================================
    // SAFE CREATION METHODS (The Right Way)
    // =====================================================

    /**
     * Assign a user as voter in an election with full integrity checks
     * 
     * @throws \Exception
     */
    public static function assignVoter(
        string $userId, 
        string $electionId, 
        ?string $assignedBy = null,
        array $metadata = []
    ): self {
        return DB::transaction(function () use ($userId, $electionId, $assignedBy, $metadata) {
            // Lock election to prevent concurrent changes
            $election = Election::lockForUpdate()->findOrFail($electionId);
            
            // Verify user is a member of the organisation (fresh check)
            $isMember = DB::table('user_organisation_roles')
                ->where('user_id', $userId)
                ->where('organisation_id', $election->organisation_id)
                ->lockForUpdate()
                ->exists();

            if (!$isMember) {
                throw new \InvalidArgumentException(
                    "User [{$userId}] is not a member of organisation [{$election->organisation_id}]"
                );
            }

            // Check for existing membership
            $existing = self::where('user_id', $userId)
                ->where('election_id', $electionId)
                ->lockForUpdate()
                ->first();

            if ($existing) {
                // Reactivate if inactive
                if ($existing->status !== 'active') {
                    $existing->update([
                        'status' => 'active',
                        'assigned_by' => $assignedBy,
                        'assigned_at' => now(),
                        'metadata' => array_merge($existing->metadata ?? [], $metadata)
                    ]);
                    return $existing;
                }
                
                throw new \InvalidArgumentException(
                    "User is already an active voter in this election"
                );
            }

            // Create new membership
            return self::create([
                'user_id' => $userId,
                'organisation_id' => $election->organisation_id,
                'election_id' => $electionId,
                'role' => 'voter',
                'status' => 'active',
                'assigned_by' => $assignedBy,
                'assigned_at' => now(),
                'metadata' => $metadata
            ]);
        }, 5); // Retry up to 5 times on deadlock
    }

    /**
     * Bulk assign voters (optimized for large imports)
     */
    public static function bulkAssignVoters(
        array $userIds, 
        string $electionId, 
        ?string $assignedBy = null
    ): array {
        return DB::transaction(function () use ($userIds, $electionId, $assignedBy) {
            $election = Election::lockForUpdate()->findOrFail($electionId);
            
            // Get all valid memberships in one query
            $validUserIds = DB::table('user_organisation_roles')
                ->whereIn('user_id', $userIds)
                ->where('organisation_id', $election->organisation_id)
                ->pluck('user_id')
                ->toArray();

            $invalidUserIds = array_diff($userIds, $validUserIds);

            // Get existing memberships
            $existing = self::where('election_id', $electionId)
                ->whereIn('user_id', $validUserIds)
                ->pluck('user_id')
                ->toArray();

            $newUserIds = array_diff($validUserIds, $existing);

            // Bulk insert new memberships
            $now = now();
            $memberships = [];
            foreach ($newUserIds as $userId) {
                $memberships[] = [
                    'id' => (string) \Str::uuid(),
                    'user_id' => $userId,
                    'organisation_id' => $election->organisation_id,
                    'election_id' => $electionId,
                    'role' => 'voter',
                    'status' => 'active',
                    'assigned_by' => $assignedBy,
                    'assigned_at' => $now,
                    'created_at' => $now,
                    'updated_at' => $now,
                    'metadata' => '{}'
                ];
            }

            if (!empty($memberships)) {
                self::insert($memberships);
            }

            return [
                'success' => count($memberships),
                'already_existing' => count($existing),
                'invalid' => count($invalidUserIds)
            ];
        });
    }

    // =====================================================
    // BUSINESS LOGIC METHODS
    // =====================================================

    /**
     * Check if voter is eligible to vote right now
     */
    public function isEligible(): bool
    {
        return $this->status === 'active' 
            && ($this->expires_at === null || $this->expires_at->isFuture());
    }

    /**
     * Mark that this voter has voted
     */
    public function markAsVoted(): void
    {
        $this->update([
            'last_activity_at' => now(),
            'status' => 'inactive' // Optional: prevent double voting
        ]);
    }

    /**
     * Remove voter from election (soft delete via status)
     */
    public function remove(string $reason = null): void
    {
        $this->update([
            'status' => 'removed',
            'metadata' => array_merge($this->metadata ?? [], [
                'removed_at' => now()->toIso8601String(),
                'removed_reason' => $reason
            ])
        ]);
    }

    // =====================================================
    // ACCESSORS & MUTATORS
    // =====================================================

    public function getDisplayNameAttribute(): string
    {
        return $this->user?->name ?? 'Unknown User';
    }

    public function getEmailAttribute(): string
    {
        return $this->user?->email ?? '';
    }

    public function getIsExpiredAttribute(): bool
    {
        return $this->expires_at !== null && $this->expires_at->isPast();
    }

    /**
     * Automatically set organisation_id from election if not provided
     */
    protected static function booted()
    {
        static::creating(function ($membership) {
            // Always ensure organisation_id matches election
            if (!$membership->organisation_id && $membership->election_id) {
                $election = Election::find($membership->election_id);
                $membership->organisation_id = $election?->organisation_id;
            }
        });
    }
}
```

```php
            300, // 5 minutes

                ->where('election_id', $electionId)
                ->where('role', 'voter')
                ->where('status', 'active')
                ->exists()
        );

     * Clear voter cache when membership changes
     */
    protected static function booted()
    {
        static::saved(function ($user) {
            Cache::delete("user.{$user->id}.voter.*");
        });
    }
```

```php
<?php

namespace App\Console\Commands;

use Illuminate\Console\Command;
use App\Models\ElectionMembership;
use Illuminate\Support\Facades\DB;
use Illuminate\Support\Facades\Log;

class ValidateElectionMemberships extends Command
{
    protected $signature = 'elections:validate-memberships';
    protected $description = 'Validate all election memberships for data integrity';

    public function handle()
    {
        $this->info('Starting election membership validation...');

        // Find memberships where user is not in organisation
        $invalid = ElectionMembership::whereNotExists(function ($query) {
            $query->select(DB::raw(1))
                ->from('user_organisation_roles')
                ->whereRaw('user_organisation_roles.user_id = election_memberships.user_id')
                ->whereRaw('user_organisation_roles.organisation_id = election_memberships.organisation_id');
        })->get();

        if ($invalid->count() > 0) {
            $this->error("Found {$invalid->count()} invalid memberships!");
            
            Log::warning('Invalid election memberships detected', [
                'count' => $invalid->count(),
                'ids' => $invalid->pluck('id')
            ]);

            // Option 1: Auto-fix (delete or mark as invalid)
            $invalid->each->delete();
            
            // Option 2: Just report
            $this->table(['ID', 'User', 'Election'], $invalid->map(fn($m) => [
                $m->id, $m->user_id, $m->election_id
            ]));
        }

        // Find memberships where election doesn't match organisation
        $mismatched = ElectionMembership::whereNotExists(function ($query) {
            $query->select(DB::raw(1))
                ->from('elections')
                ->whereRaw('elections.id = election_memberships.election_id')
                ->whereRaw('elections.organisation_id = election_memberships.organisation_id');
        })->get();

        if ($mismatched->count() > 0) {
            $this->error("Found {$mismatched->count()} mismatched memberships!");
            // Handle similarly
        }

        $this->info('Validation complete!');
    }
```

## **✅ MIGRATION FROM SINGLE ELECTION**

```php
<?php

use Illuminate\Database\Migrations\Migration;
use Illuminate\Support\Facades\DB;

class MigrateIsVoterToElectionMemberships extends Migration
{
    public function up()
    {
        DB::transaction(function () {
            // Get all users with is_voter = 1
            $voters = DB::table('users')
                ->where('is_voter', 1)
                ->get();

            foreach ($voters->groupBy('organisation_id') as $orgId => $orgVoters) {
                // Find the first election for this organisation
                $election = DB::table('elections')
                    ->where('organisation_id', $orgId)
                    ->first();

                if (!$election) {
                    // Create a default election if none exists
                    $electionId = (string) \Str::uuid();
                    DB::table('elections')->insert([
                        'id' => $electionId,
                        'organisation_id' => $orgId,
                        'title' => 'Default Election',
                        'created_at' => now(),
                        'updated_at' => now()
                    ]);
                } else {
                    $electionId = $election->id;
                }

                // Create memberships
                $memberships = [];
                $now = now();
                
                foreach ($orgVoters as $voter) {
                    $memberships[] = [
                        'id' => (string) \Str::uuid(),
                        'user_id' => $voter->id,
                        'organisation_id' => $orgId,
                        'election_id' => $electionId,
                        'role' => 'voter',
                        'status' => 'active',
                        'assigned_at' => $now,
                        'created_at' => $now,
                        'updated_at' => $now,
                        'metadata' => json_encode(['migrated_from_is_voter' => true])
                    ];
                }

                DB::table('election_memberships')->insert($memberships);
            }

            // Remove old column
            Schema::table('users', function ($table) {
                $table->dropColumn('is_voter');
            });
        });
    }
}
```
